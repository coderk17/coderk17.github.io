import os
import re
import pty
import json
import time
import fcntl
import struct
import termios
import asyncio
from json import JSONDecodeError

from fastapi.websockets import WebSocket, WebSocketDisconnect

# 文件写入状态
WRITE_FILE_ING = 0
WRITE_FILE_SUC = 1
WRITE_FILE_ERR = 2

# 文件上传状态
UPLOAD_ING = 0
UPLOAD_SUC = 1
UPLOAD_ERR = 2

# pty_stream服务实例状态
SERVICE_RUNING = 1
SERVICE_CLOSED = 2


class PtyStream:
    """终端IO流管理"""
    def __init__(self, websocket: WebSocket, event_mode: bool = True):
        self.websocket = websocket
        self.fd = self.get_pty_fd()
        self.block_size = 32 * 1024
        self.fd_event_mode = None
        # 每个分片header信息长度记录大小
        self.record_header_size = 16
        self.upload_dir = os.path.join(os.path.expanduser('~'), "upload-data")
        """ 记录每个文件上传过程中保存的中间信息，传输完成后即销毁
        self.upload_info = {
            $filename: {
                checked_filename: $checked_filename,                // 经过重名处理后的文件名
                filepath: $filepath,                                // 文件路径
                next_merge_start_index: $next_merge_start_index     // 下一次merge的起始子文件索引
                file_write_info: {                                  // 子文件写入状态
                    $block_index: $write_status 
                },
                status: $status                             // 运行状态，若成功则无此状态，若由此状态表示为异常终端此次传输
            } 
        } """
        self.upload_info = {}
        # 文件名及文件分片后子文件的文件名格式模板
        self.filepath_format = f"{self.upload_dir}/%s"
        self.subfilepath_format = f"{self.upload_dir}/.%s_%s"
        # 匹配括号
        self.re_match_bracket = re.compile(r"（\d+）")
        self.status = SERVICE_RUNING
        # 注册事件监听
        self.events = websocket.app.events
        self.events.register_read_event(self.fd, self, event_mode)

    def __enter__(self):
        return self

    def __exit__(self, exc_type, exc_value, exc_tb):
        if self.status == SERVICE_RUNING:
            self._close()

    def __del__(self):
        if self.status == SERVICE_RUNING:
            self._close()

    def get_pty_fd(self) -> int:
        """ 创建伪终端，执行预处理命令并返回伪终端对应的文件描述符

        :return: 伪终端的文件描述符
        """
        pid, fd = pty.fork()
        if pid == pty.CHILD:
            # 启动脚本，可以执行预处理命令，或更改环境
            os.execlp("/bin/bash", *["/bin/bash"])
        return fd

    def _read(self):
        """pty输出"""
        try:
            text = os.read(self.fd, self.block_size)
        except OSError as os_err:
            raise Exception(f"PtyStream:_read:{os_err}")
        except Exception as err:
            raise Exception(f"PtyStream:_read:{err}")
        else:
            return text

    def _write(self, data: str):
        """pty输入"""
        os.write(self.fd, bytes(data, encoding="utf-8"))

    def _close(self):
        """pty关闭"""
        self.status = SERVICE_CLOSED
        os.close(self.fd)
        self.events.unregister(self.fd)
        raise WebSocketDisconnect("终端关闭，断开连接")

    async def on_read(self):
        """读取pty stdout并通过websocket发送给客户端"""
        text = self._read()
        await self.websocket.send_text(text)
        if text[-6:] == b'exit\r\n':
            close_msg = "您已经断开连接，输入已关闭，请尝试另一次连接。"
            await self.websocket.send_text(bytes(close_msg, encoding="utf-8"))
            self._close()

    def clear_stdout_buffer(self):
        """清空pty stdout缓冲区"""
        self._read()

    def exec_command(self, command: str):
        """ pty执行命令行

        :param command: 要执行的命令行字符串
        :return:
        """
        try:
            os.write(self.fd, bytes(f"{command}\r", encoding="utf-8"))
        except Exception as err:
            raise Exception(f"exec_command:{command}:{err}")

    def resize_window(self, rows: int, cols: int):
        """ pty终端大小设置

        :param rows: 行数
        :param cols: 列数
        :return:
        """
        width, height = cols, rows
        TIOCSWINSZ = getattr(termios, 'TIOCSWINSZ', -2146929561)
        s = struct.pack('HHHH', width, height, 0, 0)
        fcntl.ioctl(self.fd, TIOCSWINSZ, s)

    async def process_pty_recv(self, message: str):
        """ 传入pty数据的处理逻辑

        :param message: 传入pty的数据
        :return:
        """
        try:
            if self.status != SERVICE_RUNING:
                raise WebSocketDisconnect("终端关闭，断开连接")
            try:
                msg = json.loads(message)
            except JSONDecodeError:
                raise Exception("process_pty_recv:%s" % message)

            resize = msg.get('resize')
            if resize and len(resize) == 2:
                try:
                    self.resize_window(*resize)
                except Exception as err:
                    raise Exception("process_pty_recv:resize:%s" % err)

            data = msg.get('data')
            if data and isinstance(data, str):
                self._write(data)
        except WebSocketDisconnect as disconnect_err:
            raise disconnect_err
        except Exception as err:
            raise Exception("process_pty_recv:%s" % err)

    def _check_filename(self, filename: str, max_dup_file_nums: int = 5) -> str:
        """ 检测文件名是否已存在，若已存在，则在该文件名中加上重名标识符

        若上传文件夹中已包含a.txt，再次上传a.txt，会生成一个a（1）.txt文件，
        若再次上传a.txt会生成a（2）.txt，可更改max_dup_file_nums控制标识符
        最大值，默认为5

        :param filename: 检测的文件名
        :param max_dup_file_nums: 添加的标识符最大值
        :return: 检测处理完成后的文件名
        """
        for i in range(max_dup_file_nums + 1):
            exist_status = os.path.exists(self.filepath_format % filename)
            if exist_status:
                if i == max_dup_file_nums:
                    raise Exception("Duplicated file is too much, please change file name.")
                if len(filename) > 1024:
                    raise Exception("Filename's length is too long.")

                if self.re_match_bracket.search(filename):
                    filename = self.re_match_bracket.sub("", filename)
                if "." in filename:
                    filename_split = filename.split(".")
                    filename = ".".join(filename_split[:-1]) + f"（{i + 1}）" + "." + filename_split[-1]
                else:
                    filename = filename + f"（{i + 1}）"
            else:
                break
        return filename

    def _write_file(self, filepath: str, data: bytes) -> int:
        """ 指定文件路径，写入数据，并返回写入成功状态

        :param filepath: 文件路径
        :param data: 数据
        :return: 成功状态，WRITE_FILE_ERR表示写入失败，WRITE_FILE_SUC表示写入成功
        """
        try:
            with open(filepath, "wb+") as f:
                f.write(data)
        except Exception as err:
            status = WRITE_FILE_ERR
        else:
            status = WRITE_FILE_SUC
        return status

    async def _merge_sub_file(self, ori_filename: str, start_index: int, end_index: int, timeout: int = 5):
        """ 合并上传文件的切片子文件

        :param ori_filename: 客户端上传的文件原名
        :param start_index: 处理的起始子文件索引
        :param end_index: 处理的终止子文件索引
        :param timeout: 处理的超时时间
        :return:
        """
        try:
            upload_info = self.upload_info[ori_filename]
            file_write_info = upload_info["file_write_info"]
            filepath = upload_info["filepath"]
            checked_filename = upload_info["checked_filename"]
            check_indexes = list(range(start_index, end_index + 1))
            starttime = time.time()
            while (time.time() - starttime) < timeout:
                new_check_indexes = []
                for i in check_indexes:
                    if i not in file_write_info:
                        raise Exception("上传文件传输包缺失：filename:%s,start_index:%s,end_index:%s" % (ori_filename, start_index, end_index))
                    if file_write_info[i] == WRITE_FILE_ERR:
                        raise Exception("上传文件传输包写入失败：filename:%s,start_index:%s,end_index:%s" % (ori_filename, start_index, end_index))
                    if file_write_info[i] == WRITE_FILE_ING:
                        new_check_indexes.append(i)
                check_indexes = new_check_indexes
                if len(check_indexes) == 0:
                    break
                await asyncio.sleep(0.1)
            if len(check_indexes) != 0:
                raise Exception("上传文件处理超时，filename: %s" % ori_filename)
            for i in range(start_index, end_index + 1):
                subfilepath = self.subfilepath_format % (checked_filename, i)
                command_merge = f"cat {subfilepath} >> {filepath}"
                self.exec_command(command_merge)
                command_del_sub = f"rm {subfilepath}"
                self.exec_command(command_del_sub)
        except Exception as err:
            raise Exception("_merge_sub_file:%s" % err)

    async def process_recv_file_bytes(self, message: bytes):
        """ 文件上传中获取数据流的处理程序

        :param message: 文件上传中获取的数据流
        :return:
        """
        try:
            if self.status != SERVICE_RUNING:
                raise WebSocketDisconnect("终端关闭，断开连接")
            # 当子文件个数达到merge_block_size时执行merge到父文件操作
            MERGE_BLOCK_SIZE= 10
            # FIXME 文件传输如果有阻塞会对终端操作产生影响，改为非阻塞文件传输，目前分片大小为1M上传无阻塞
            # 目前只有上传文件功能，没有上传文件夹功能，后期如果需要可以加上
            header_info_end_index = int(''.join([str(x) for x in message[:self.record_header_size]]))
            header_info = json.loads(message[self.record_header_size: header_info_end_index])
            ori_filename = header_info["create_file"]
            block_index = header_info["block_index"]
            end_signal = header_info.get("end_signal", False)
            message = message[header_info_end_index:]
            if block_index == 0:
                # 第一个数据块检查文件名重名并创建文件，同时初始化传输过程数据（upload_info）
                checked_filename = self._check_filename(ori_filename, max_dup_file_nums=5)
                filepath = self.filepath_format % checked_filename
                self.exec_command(command=f"touch {filepath}")

                # TODO 同时上传两个同名文件，需要前端在header里加一个文件唯一标识符字段，
                # 可以用文件内容哈希值，可以实现同时上传，不过目前来看没有必要加上
                if ori_filename in self.upload_info:
                    raise Exception("正在上传一个同名文件，不能同时上传两个同名文件")
                upload_info = self.upload_info.setdefault(ori_filename, {})
                upload_info["checked_filename"] = checked_filename
                upload_info["filepath"] = filepath
                upload_info["file_write_info"] = {}
                upload_info["next_merge_start_index"] = 0

            upload_info = self.upload_info[ori_filename]
            checked_filename = upload_info["checked_filename"]
            file_write_info = upload_info["file_write_info"]
            # 接收信息，写入子文件，并记录写入状态
            file_write_info.setdefault(block_index, WRITE_FILE_ING)
            subfilepath = self.subfilepath_format % (checked_filename, block_index)
            write_status = self._write_file(subfilepath, message)
            file_write_info[block_index] = write_status

            # merge子文件内容到结果文件中
            # 可优化为merge_block_size个线程执行，不影响总顺序，
            # 因为每次合并时会给这merge_block_size个文件检查存在性并排序
            next_merge_start_index = upload_info["next_merge_start_index"]
            if (((block_index + 1) % MERGE_BLOCK_SIZE == 0)
                or end_signal):
                await self._merge_sub_file(
                    ori_filename,
                    start_index=next_merge_start_index,
                    end_index=block_index
                )
                upload_info["next_merge_start_index"] = block_index + 1

            if end_signal:
                del(self.upload_info[ori_filename])
        except WebSocketDisconnect as disconnect_err:
            raise disconnect_err
        except Exception as err:
            # TODO 报错时保存其他上下文，如已传输文件内容哈希，各子文件名及哈希，多请求几个包校验前端传输分片内容是否可信任
            if ori_filename in self.upload_info:
                self.upload_info[ori_filename]["status"] = UPLOAD_ERR
            resp = {
                "create_file": header_info["create_file"],
                "block_index": block_index,
                "block_end_signal": end_signal,
                "msg": str(err),
                "status_code": 500
            }
            raise Exception("process_recv_file_bytes:%s" % err)
        else:
            resp = {
                "create_file": header_info["create_file"],
                "block_index": block_index,
                "block_end_signal": end_signal,
                "msg": "OK",
                "status_code": 200
            }
        finally:
            await self.websocket.send_text(json.dumps(resp))
